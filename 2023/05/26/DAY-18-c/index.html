#include

using namespace std;

class Person{
protected:
string sex;
int age;
string name;
public:
Person(string nam,int ag,string se):name(nam),age(ag),sex(se){}
void ShowPerson() {
cout << “name: “ << name << endl;
cout << “age: “ << age << endl;
cout << “sex: “ << sex << endl;
}
};

class Student : virtual public Person {
protected:
int studentNumber;
public:
Student(string nam,int ag,string se,int num):Person(nam,ag,se )
{
studentNumber=num;
}
void ShowStudent(){cout<<”studentNumber: “<<studentNumber<<endl;}
};
class Employee : virtual public Person {
protected:
string career;
public:
Employee(string nam,int ag,string se,string caree):Person(nam,ag,se ){career=caree;}
void ShowCareer()
{
cout<<”career: “<<career<<endl;
}
};
class PartTime_Graduate_Student : public Employee, public Student{
private:
int wage;
public:
PartTime_Graduate_Student(string nam,int ag,string se,int num,string caree,int wag):Person(nam,ag,se),Student(nam,ag,se,num),Employee(nam,ag,se,caree),wage(wag){};

void showPart() {
    ShowPerson();
    ShowCareer();
    ShowStudent();
};

};
