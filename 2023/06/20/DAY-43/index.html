题目
实验九：多态性
实验目的：
（1）了解多态性的概念
（2）了解虚函数的作用和使用方法
（3）了解纯虚函数和抽象类的概念和使用方法
实验内容：
声明类Shape以及它的3个派生类：Circle、Square、Rectangle，
用虚函数分别计算几种图形的面积、周长，编写程序，建立基
类指针数组，每个指针均指向一个派生类对象，利用基类指针
遍历元素对象，求取所有图形面积之和。请绘制UML类图

#include
const double pi= 3.1415926;
using namespace std;

class Shape {
public:
Shape(){
}
~Shape(){
}
virtual double perimeter(){
}
virtual double area(){
}
};
class Circle:public Shape{
private:
double radius;
public:
Circle(double r):radius(r){
}
~Circle(){
}
double perimeter(){
return 2piradius;
}
double area(){
return piradiusradius;
}
};
class Square:public Shape{
private:
double side;
public:
Square(double sid):side(sid){
}
~Square(){

}
double perimeter(){
return side*4;
}
double area(){
    return side*side;
}

};
class Rectangle:public Shape{
private:
double length;
double width;
public:
Rectangle(double len,double wid){
len= length;
wid= width;
}
~Rectangle(){
}
double perimeter(){
return (length+width)2;
}
double area(){
return lengthwidth;
}
};

int main()
{
//Shape 有三个派生类
//创造一个基类指针数组
Shape *nihao[3];
nihao[0]=new Circle(2);

nihao[2]=new Rectangle(3,2)
cout << nihao[0].area();
cout << nihao[0].perimeter();
return 0;
}
自己写的多态性代码
1.没有进行delete操作,没有进行delete操作的话会导致内存泄漏，会导致程序内存占用增加，造成额外的内存压力
2.对于指针对象数组的函数引用要用箭头进行指向->
