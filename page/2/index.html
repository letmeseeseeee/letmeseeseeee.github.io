DAY 47

protected 在子类定义和当前类定义中可以访问

private 只在当前类定义中可以访问

public 在任何地方中都可以访问

class GoldenBox{
public:
GoldenBox(int);
void show(Box & );
private:
int weight;
};
class Box {
public:
Box(int ,int ,int );
friend void GoldenBox::show(Box & );
private:
int width;
int length;
int height;
};
friend 数据类型 classtype::functionname

include

using namespace std;
class Box; //pre-reference declaration
class GoldenBox{
public:
GoldenBox(int);
void show(Box & );
private:
int weight;
};
class Box {
public:
Box(int ,int ,int );
friend GoldenBox;
private:
int width;
int length;
int height;
}
Box 有一个友元类GoldenBox

友元类（Friend Class）是一种特殊的类关系，允许一个类的成员函数或友元函数访问另一个类的私有成员。友元类可以访问其它类的私有成员，包括私有变量和私有函数，而不需要通过公有接口进行访问。

但是使用友元类需要谨慎

#include

using namespace std;

class Person{
private:
string name;
char sex;
public:
Person(string a,char s):name(a),sex(s){
}
virtual ~Person{

}
virtual void info(){
cout << "名字是" << name ;
cout << "性别是" << sex;
}

};
class student:virtual public Person{
private:
int number;
int grade;
public:
student(int a,int g ):number(a),grade(g){
}
~student(){
}
void info(){
Person.info();
cout << “学号是” << number<< endl;
cout << “年级是” << grade <<endl;
}
};
class Employee:virtual public Person{
private:
int worknum;
double wage;
string job;
public:
Employee(int a,double b , string c):worknum(a),wage(b),job(c){
}
~Employee(){
}
void info (){
Person.info();
cout << “工号是” <<worknum<< endl;
cout << “工资是” <<wage <<endl;
cout << “工作是” <<job << endl ;
}

};
class PartTime_Graduate_Student:public Employee,public student{
private:

student.info();
Employee.info();

}

};
int main()
{
PartTime_Graduate_Student nihao(“李伯辉”,”男”,19,1,1,10000,”程序员”);
nihao.info();

return 0;

}

首先，在派生类引用基类的成员函数的时候需要用::而不是.

其次在两个派生类的构造函数中初始化错误，应该引用虚基类的构造函数

并且在最小的那个派生类中的构造函数初始化成员数值也出现了错误

PartTime_Graduate_Student(string a, char b ,int d ,int e ,int f ,double g ,string h):Person(a,b),student(d,e),Employee(f,g,h)
{
}
上面是重写的一个派生类，但仍然有错误

再用student和Employee进行构造函数对数值进行初始化的时候要把之前的虚基类中的两个私有成员数值也加上

PartTime_Graduate_Student(string a, char b ,int d ,int e ,int f ,double g ,string h):Person(a,b),student(a,b,d,e),Employee(a,b,f,g,h)
{
}
修改好之后的函数

并且，由于b是char类型所以男应该用单引号括‘c’，而不是双引号”c”

一开始输出错误了，但是加上endl又正常输出了不知道为什么

如果不使用 << endl，输出的内容将保持在输出缓冲区中，直到达到缓冲区刷新的条件（例如缓冲区填满或遇到换行符等），才会将内容显示在终端上。因此，如果不加 << endl，输出可能会被缓冲而不立即显示。

实验内容：
声明类Shape以及它的3个派生类：Circle、Square、Rectangle，
用虚函数分别计算几种图形的面积、周长，编写程序，建立基
类指针数组，每个指针均指向一个派生类对象，利用基类指针
遍历元素对象，求取所有图形面积之和。请绘制UML类图

#include
const double pi= 3.1415926;
using namespace std;

class Shape {
public:
Shape(){
}
~Shape(){
}
virtual double perimeter(){
}
virtual double area(){
}
};
class Circle:public Shape{
private:
double radius;
public:
Circle(double r):radius(r){
}
~Circle(){
}
double perimeter(){
return 2piradius;
}
double area(){
return piradiusradius;
}
};
class Square:public Shape{
private:
double side;
public:
Square(double sid):side(sid){
}
~Square(){

}
double perimeter(){
return side*4;
}
double area(){
    return side*side;
}

};
class Rectangle:public Shape{
private:
double length;
double width;
public:
Rectangle(double len,double wid){
len= length;
wid= width;
}
~Rectangle(){
}
double perimeter(){
return (length+width)2;
}
double area(){
return lengthwidth;
}
};

int main()
{
//Shape 有三个派生类

return 0;
}
